Function GetValueFromDPRFile(ByVal wbPath As String, _
    DPRName As String, wsName As String, cellRef As String) As Variant
Dim arg As String
Dim bc As String
bc = Chr(34) + Chr(34)
    GetValueFromDPRFile = ""
    If Right(wbPath, 1) <> "\" Then wbPath = wbPath & "\"
    If Dir(wbPath & "\" & DPRName) = "" Then Exit Function
    arg = "'" & wbPath & "[" & DPRName & "]" & _
        wsName & "'!" & Range(cellRef).Address(True, True, xlR1C1)
    On Error Resume Next
    'IF(ISBLANK(OutputDW!G8),"",OutputDW!G8)
    arg = "IF(ISBLANK(" & arg & ")," & bc & "," & arg & ")"
    'MsgBox (arg)
    GetValueFromDPRFile = ExecuteExcel4Macro(arg)
End Function

Sub AutoDPRname(ByVal dt As Date)

    Set rg1 = wk.Range("B1").CurrentRegion
    numDW = rg1.Rows.Count
    
    Dim sLeft As String, ext As String, sRight As String, DPRAlias As String, DPRFileAlias As String, DPRFileName As String, DFromat As String
    
    dt = Format(dt, "[$-en-US]dd/mm/yyyy;@")
    
    '1DPR Alias   2DPR_File_Alias   3ext 4Format  5L   6R   7pref    8suf

    For i = LBound(arrListDPR, 1) + 1 To UBound(arrListDPR, 1)
    DPRAlias = arrListDPR(i, 1)
    DPRFileAlias = arrListDPR(i, 2)
    ext = "." & arrListDPR(i, 3)
    DFromat = arrListDPR(i, 4)
    sLeft = arrListDPR(i, 5)
    sRight = arrListDPR(i, 6)
        'WorksheetFunction.Text
    If (sLeft <> "") Then
    sLeft = Replace(sLeft, "0", "")
    DPRFileName = WorksheetFunction.Text(dt, DFromat) & sLeft & DPRFileAlias & ext
    End If
    If (sRight <> "") Then
    sRight = Replace(sRight, "0", "")
    DPRFileName = DPRFileAlias & sRight & WorksheetFunction.Text(dt, DFromat) & arrListDPR(i, 8) & ext
    End If
      
    If (DPRAlias = "RAPPORT MLD") Then
    DPRFileName = DPRFileAlias & sRight & iDayNumber & ext
    End If
    
    
    If (DPRAlias = "HSD Daily Prod") Then
    DPRFileName = DPRFileAlias & sRight & Format(dt, DFromat) & arrListDPR(i, 8) & ext
    End If
   
    
    'Debug.Print DPRAlias, DPRFileName
    sLeft = ""
    sRight = ""
    
    
    
    
    
    dicDPRList.Add DPRAlias, DPRFileName
    
    
    
    Next i






End Sub
